\int_L \vec F \cdot d\vec r&=\int_L (P(x, y), Q(x, y)) \cdot(dx, dy)\\
&=\int_L Pdx +Qdy
\end{align}
$$

$$
\int_L P(x, y)dx + Q(x, y)dy = \int_\alpha^\beta \{P[\phi(t),\psi(t)]\phi\prime(t)+Q[\phi(t), \psi(t)]\psi\prime(t)\}dt
